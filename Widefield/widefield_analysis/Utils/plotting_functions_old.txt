def fdr_p_values(p_values, indicies, image_height, image_width, alpha=0.05):
    p_values = np.nan_to_num(p_values, nan=1)
    p_values = np.reshape(p_values, image_height * image_width)
    brain_p_values = p_values[indicies]
    rejected, corrected_p = fdrcorrection(brain_p_values, alpha=alpha)
    p_values[indicies] = corrected_p
    p_values = np.reshape(p_values, (image_height, image_width))
    return p_values



def compare_regressors(regressor_1, regressor_2, save_directory, start_window, stop_window):

    # Baseline Correct Regressor
    regressor_1 = baseline_correct_regressors(regressor_1)
    regressor_2 = baseline_correct_regressors(regressor_2)

    # Get Mean Regressors
    mean_regressor_1 = np.mean(regressor_1, axis=0)
    mean_regressor_2 = np.mean(regressor_2, axis=0)

    # Get Signficance
    t_stats, p_values = stats.ttest_rel(regressor_1, regressor_2, axis=0)
    print("p_values", np.shape(p_values))

    # Should be size (n_timepoints, image_height, image_width)
    n_timepoints, image_height, image_width = np.shape(mean_regressor_1)
    print("mean regressor", np.shape(mean_regressor_1))

    # Get Colourmaps
    cmap = Mean_Activity_Utils.get_musall_cmap()

    magnitude = 1
    norm = Normalize(vmin=-magnitude, vmax=magnitude)
    colourmap = ScalarMappable(cmap=cmap, norm=norm)

    diff_magnitude = 0.3
    diff_norm = Normalize(vmin=-diff_magnitude, vmax=diff_magnitude)
    diff_colourmap = ScalarMappable(cmap=cmap, norm=diff_norm)

    # Get Atlas Outlines
    atlas_pixels = Mean_Activity_Utils.get_atlas_outline_pixels()

    # Load Mask
    indicies, image_height, image_width = Mean_Activity_Utils.load_tight_mask()

    # Get Background Pixels
    #background_pixels = Mean_Activity_Utils.get_background_pixels(indicies, image_height, image_width)
    background_pixels = Mean_Activity_Utils.get_full_outlines()

    # Get X Values
    x_values = list(range(start_window, stop_window))
    x_values = np.multiply(x_values, 37)

    for timepoint_index in range(n_timepoints):

        # Create Figure
        figure_1 = plt.figure(figsize=(20,5))
        regressor_1_axis = figure_1.add_subplot(1,4,1)
        regressor_2_axis = figure_1.add_subplot(1,4,2)
        difference_axis = figure_1.add_subplot(1,4,3)
        sig_diff_axis = figure_1.add_subplot(1,4,4)

        # Get Timepoint Data
        regressor_1_timepoint_data = mean_regressor_1[timepoint_index]
        regressor_2_timepoint_data = mean_regressor_2[timepoint_index]
        difference = np.subtract(regressor_1_timepoint_data, regressor_2_timepoint_data)

        timepoint_p_values = p_values[timepoint_index]
        timepoint_p_values = fdr_p_values(timepoint_p_values, indicies, image_height, image_width)
        timepoint_sig = np.where(timepoint_p_values < 0.05, 1, 0)
        sig_difference = np.multiply(difference, timepoint_sig)

        # Convert To Image
        regressor_1_timepoint_data = create_image(regressor_1_timepoint_data, colourmap, atlas_pixels, background_pixels)
        regressor_2_timepoint_data = create_image(regressor_2_timepoint_data, colourmap, atlas_pixels, background_pixels)
        difference = create_image(difference, diff_colourmap, atlas_pixels, background_pixels)
        sig_difference = create_image(sig_difference, diff_colourmap, atlas_pixels, background_pixels)

        # Remove Olfactory Bulb
        regressor_1_timepoint_data = regressor_1_timepoint_data[58:]
        regressor_2_timepoint_data = regressor_2_timepoint_data[58:]
        difference = difference[58:]
        sig_difference = sig_difference[58:]
        
        # Plot Data
        regressor_1_handle = regressor_1_axis.imshow(regressor_1_timepoint_data, vmin=-magnitude, vmax=magnitude, cmap=cmap)
        regressor_2_handle = regressor_2_axis.imshow(regressor_2_timepoint_data, vmin=-magnitude, vmax=magnitude, cmap=cmap)
        diff_handle = difference_axis.imshow(difference, vmin=-diff_magnitude, vmax=diff_magnitude, cmap=cmap)
        sig_diff_handle = sig_diff_axis.imshow(sig_difference, vmin=-diff_magnitude, vmax=diff_magnitude, cmap=cmap)

        # Remove Axis
        regressor_1_axis.axis('off')
        regressor_2_axis.axis('off')
        difference_axis.axis('off')
        sig_diff_axis.axis('off')

        # Add Colourbar
        colourbar_ticks = np.around(np.linspace(start=-magnitude, stop=magnitude, num=5), 2)
        diff_colourbar_ticks = np.around(np.linspace(start=-diff_magnitude, stop=diff_magnitude, num=5), 2)
        figure_1.colorbar(regressor_1_handle, ax=regressor_1_axis, orientation='vertical', ticks=colourbar_ticks, fraction=0.046, pad=0.04)
        figure_1.colorbar(regressor_2_handle, ax=regressor_2_axis, orientation='vertical', ticks=colourbar_ticks, fraction=0.046, pad=0.04)
        figure_1.colorbar(diff_handle, ax=difference_axis, orientation='vertical', ticks=diff_colourbar_ticks, fraction=0.046, pad=0.04)
        figure_1.colorbar(sig_diff_handle, ax=sig_diff_axis, orientation='vertical', ticks=diff_colourbar_ticks, fraction=0.046, pad=0.04)

        # Set Title
        figure_1.suptitle(str(x_values[timepoint_index]) + "ms")

def get_roi_mean(regressor, atlas, roi_list):

    # Load Atlas
    indicies, image_height, image_width = Mean_Activity_Utils.load_tight_mask()

    # Get Brain Pixels
    n_mice, n_timepoints, image_height, image_width = np.shape(regressor)

    regressor = regressor[:, :, indicies]
    regressor = np.squeeze(regressor)

    print("regressor", np.shape(regressor))
    # Get ROI Pixels
    roi_pixels = Mean_Activity_Utils.get_roi_pixels(atlas, roi_list)

def baseline_correct_regressors(regressor):

    baseline_corrected_regressor = []
    for mouse in regressor:
        mouse_baseline = mouse[0:14]
        mouse_baseline = np.mean(mouse_baseline, axis=0)
        mouse = np.subtract(mouse, mouse_baseline)
        baseline_corrected_regressor.append(mouse)
    baseline_corrected_regressor = np.array(baseline_corrected_regressor)
    return baseline_corrected_regressor

def get_session_roi_rt_mean(session, output_root, rt_window_start, rt_window_stop, atlas, roi_list):

    predicted_file = os.path.join(output_root, session, "RT_Window_Predictions", file_name + "_predicted.npy")
    real_file = os.path.join(output_root, session, "RT_Window_Predictions", file_name + "_real.npy")
    mousecam_file =  os.path.join(output_root, session, "RT_Window_Predictions", file_name + "_mousecam_only.npy")

    print(session)
    print(predicted_file)
    if os.path.exists(predicted_file):

        print("exists")
        rt_window_data_predicted = np.load(predicted_file)
        rt_window_data_real = np.load(real_file)
        rt_window_data_mousecam = np.load(mousecam_file)

        roi_mean_activity_predicted = get_roi_mean_single(rt_window_data_predicted, atlas, roi_list)
        roi_mean_activity_real = get_roi_mean_single(rt_window_data_real, atlas, roi_list)
        roi_mean_activity_mousecam = get_roi_mean_single(rt_window_data_mousecam, atlas, roi_list)

    else:
        print("does not exist")
        return None, None, None




def get_group_results(session_list, output_root, rt_window_start, rt_window_stop, atlas, roi_list):

    group_list_real = []
    group_list_predicted = []
    group_list_mousecam = []

    for mouse in session_list:
        mouse_list_real = []
        mouse_list_predicted = []
        mouse_list_mousecam = []

        for session in mouse:
            session_activity_real, session_activity_predicted, session_activity_mousecam = get_session_roi_rt_mean(session, output_root, rt_window_start, rt_window_stop, atlas, roi_list)
            if isinstance(session_activity_real, np.ndarray):
                mouse_list_real.append(session_activity_real)
                mouse_list_predicted.append(session_activity_predicted)
                mouse_list_mousecam.append(session_activity_mousecam)

        if len(mouse_list_real) == 1:
            print("mouse had 1 session")
            mouse_mean_real = mouse_list_real[0]
            mouse_mean_predicted = mouse_list_predicted[0]
            mouse_mean_mousecam = mouse_list_mousecam[0]

            group_list_real.append(mouse_mean_real)
            group_list_predicted.append(mouse_mean_predicted)
            group_list_mousecam.append(mouse_mean_mousecam)

        elif len(mouse_list_real) > 1:
            print("mouse had > 1 session")
            mouse_mean_real = np.mean(mouse_list_real, axis=0)
            mouse_mean_predicted = np.mean(mouse_list_predicted, axis=0)
            mouse_mean_mousecam = np.mean(mouse_list_mousecam, axis=0)

            group_list_real.append(mouse_mean_real)
            group_list_predicted.append(mouse_mean_predicted)
            group_list_mousecam.append(mouse_mean_mousecam)

    group_list_real = np.array(group_list_real)
    group_list_predicted = np.array(group_list_predicted)
    group_list_mousecam = np.array(group_list_mousecam)

    return group_list_real, group_list_predicted, group_list_mousecam


def plot_roi_by_rt_window(session_list, output_root, start_window, stop_window, rt_window_start_list, rt_window_stop_list, roi_list):

    # Load Atlas
    atlas = Mean_Activity_Utils.load_atlas()
    atlas = np.abs(atlas)

    # Create Figure
    figure_1 = plt.figure()
    real_axis = figure_1.add_subplot(1,3,1)
    predicted_axis = figure_1.add_subplot(1, 3, 2)
    mousecam_axis = figure_1.add_subplot(1, 3, 3)
    x_values = list(range(start_window, stop_window))
    x_values = np.multiply(x_values, 37)
    rt_colourmap = plt.get_cmap('plasma')

    # Iterate Through Each Window
    n_rt_windows = len(rt_window_start_list)
    for rt_window_index in range(n_rt_windows):

        # Get Window Start and Stop
        rt_window_start = rt_window_start_list[rt_window_index]
        rt_window_stop = rt_window_stop_list[rt_window_index]

        # Get Window Data
        group_list_real, group_list_predicted, group_list_mousecam = get_group_results(session_list, output_root, rt_window_start, rt_window_stop, atlas, roi_list)

        if len(group_list_real) > 0:

            # Get Window Stop Frame
            rt_window_stop_frame = np.abs(start_window) + int(rt_window_stop / 37)

            # Take Only Until Lick
            group_list_real = group_list_real[:, 0:rt_window_stop_frame]
            group_list_predicted = group_list_predicted[:, 0:rt_window_stop_frame]
            group_list_mousecam = group_list_mousecam[:, 0:rt_window_stop_frame]
            truncated_x = x_values[0:rt_window_stop_frame]

            # Get Mean and SEM
            real_mean, real_lower, real_upper = Mean_Activity_Utils.get_mean_sd(group_list_real)
            predicted_mean, predicted_lower, predicted_upper = Mean_Activity_Utils.get_mean_sd(group_list_predicted)
            mousecam_mean, mousecam_lower, mousecam_upper = Mean_Activity_Utils.get_mean_sd(group_list_mousecam)

            # Get Window Colour
            window_colour = rt_colourmap(float(rt_window_index) / n_rt_windows)

            # Plot Means
            real_axis.plot(truncated_x, real_mean, c=window_colour, linestyle='dashed')
            predicted_axis.plot(truncated_x, predicted_mean, c=window_colour)
